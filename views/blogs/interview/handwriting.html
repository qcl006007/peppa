function findParentIds(id, list) {
  let result = [];
  list.forEach(tree => {
    let temp = [];
    dfs(tree, id, temp);
    if (temp.length > 0) {
      result.push(temp);
    }
  });
  return result.length>1? result: result[0];

}

function dfs(root, id, result) {
  if (root.id === id) {
    result.unshift(root.id);
    return true;
  }
  if (root.children == undefined) {
    return false;
  }
  for (let child of root.children) {
    if (dfs(child, id, result)) {
      result.unshift(root.id);
      return true;
    }
  }
  return false;
}
